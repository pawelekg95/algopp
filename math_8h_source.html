157template <typename ArgumentType>
158double root(ArgumentType number, unsigned int base)
159{
160 double precision = epsilon() / 10000.0;
161 auto multiplication = [](double currentNumber, unsigned int base) {
162 double token{currentNumber};
163 for (unsigned int i = 1; i < base; i++)
164 {
165 token = token * currentNumber;
166 }
167 return token;
168 };
169 const double cMultiplier = 0.5;
170 int maxSpins = 100000;
171
172 auto increase =
173 [&multiplication, &base, &precision, &maxSpins](double& token, const double& threshold, double& step) {
174 while (multiplication(token, base) - threshold < precision)
175 {
176 if (maxSpins < 0)
177 {
178 break;
179 }
180 token += step;
181 maxSpins--;
182 }
183 };
184
185 auto decrease =
186 [&multiplication, &base, &precision, &maxSpins](double& token, const double& threshold, double& step) {
187 while (multiplication(token, base) - threshold > precision)
188 {
189 if (maxSpins < 0)
190 {
191 break;
192 }
193 token -= step;
194 maxSpins--;
195 }
196 };
197
198 double token = epsilon();
199 double jump = cMultiplier * number;
200
201 while (abs(multiplication(token, base) - number) > precision)
202 {
203 if (maxSpins < 0)
204 {
205 break;
206 }
207 auto multiplied = multiplication(token, base);
208 if (multiplied - number > precision)
209 {
210 jump = jump * cMultiplier;
211 decrease(token, number, jump);
212 }
213 else if (multiplied - number < precision)
214 {
215 jump = jump * cMultiplier;
216 increase(token, number, jump);
217 }
218 maxSpins--;
219 }
220 return token;
221}

260template <typename ArgumentType, typename PowerType>
261long double pow(ArgumentType number, PowerType power)
262{
263 Fraction pwrFraction(power);
264 if (power < 0 && int(power) % 2 == 0 && !pwrFraction)
265 {
